Here is a simple **HTML program** to display an image using Bootstrap with different styles.

---


<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Bootstrap Image Example</title>
    <!-- Bootstrap CSS -->
    <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/css/bootstrap.min.css" rel="stylesheet">
</head>
<body>

    <div class="container mt-4">
        <h2 class="text-center">Bootstrap Image Example</h2>

        <!-- Basic Image -->
        <h4>Basic Image:</h4>
        <img src="https://via.placeholder.com/300" alt="Sample Image" class="img-fluid">

        <!-- Rounded Image -->
        <h4 class="mt-3">Rounded Image:</h4>
        <img src="https://via.placeholder.com/300" alt="Sample Image" class="img-fluid rounded">

        <!-- Circle Image -->
        <h4 class="mt-3">Circular Image:</h4>
        <img src="https://via.placeholder.com/300" alt="Sample Image" class="img-fluid rounded-circle">

        <!-- Thumbnail Image -->
        <h4 class="mt-3">Thumbnail Image:</h4>
        <img src="https://via.placeholder.com/300" alt="Sample Image" class="img-thumbnail">
    </div>

    <!-- Bootstrap JS -->
    <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/js/bootstrap.bundle.min.js"></script>
</body>
</html>
